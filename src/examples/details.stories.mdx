# Details

The following example shows how a detailed view could look like, e.g. from a user.

export const Template = ({ title }) => {
  return (
    <ContentArea>
      <Section>
        <SectionHeader title={title} />
        <SectionContainer>
          <SectionContent layout={SectionContentLayout.oneColumnGrid}>
            <DescriptionItem content="John Doe" title="Name" />
            <DescriptionItem content="info@aboutbits.it" title="Email" />
            <DescriptionItem content="admin" title="Role" />
            <DescriptionItem content="English" title="Language" />
            <DescriptionItem
              content="Java, Python, C++, Rust, PHP, JavaScript, Springboot, Laravel, React, Vue, TypeScript"
              title="Languages and Frameworks"
            />
            <DescriptionItem
              content="Teamwork, Time management"
              title="Soft skills"
            />
            <DescriptionItem
              content="John is a software engineer from Bolzano, Italy. In addition to his technical expertise, John is a team player and is known for his ability to work well with others. "
              title="Bio"
            />
          </SectionContent>
        </SectionContainer>
      </Section>
    </ContentArea>
  )
}

<Canvas>
  <Story name="Default" args={{ title: 'Detailed user view' }}>
    {Template.bind({})}
  </Story>
</Canvas>
