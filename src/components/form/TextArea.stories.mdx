import { yupResolver } from '@hookform/resolvers/yup'
import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs'
import { action } from '@storybook/addon-actions'
import { useForm } from 'react-hook-form'
import * as Yup from 'yup'
import { Mode } from '../types'
import { Form } from './Form'
import { TextArea } from './TextArea'
import { FormVariant } from './types'

<Meta
  title="Components/Form/TextArea"
  component={TextArea}
  decorators={[
    (Story) => {
      const resolver = yupResolver(
        Yup.object().shape({
          description: Yup.string().required('Required'),
        })
      )
      const form = useForm({
        resolver,
        defaultValues: { description: '' },
        mode: 'onTouched',
      })
      return (
        <Form form={form} onSubmit={action('onSubmit')}>
          <Story />
        </Form>
      )
    },
  ]}
/>

# TextArea

This component can be used to create textarea fields. This input already has a label and handles error messages in `React Hook Form` Forms.

export const Template = ({ ...args }) => <TextArea {...args} />
