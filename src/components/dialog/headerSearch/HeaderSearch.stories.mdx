# HeaderSearch

This component contains the logic for the search functionality of the dialog header.

export const Template = ({ ...args }) => {
  const [text, setText] = useState('')
  const clearSearch = () => setText('')
  return (
    <NavigationProvider>
      <HeaderSearch
        {...args}
        text={text}
        setText={setText}
        clearSearch={clearSearch}
        stopSearch={clearSearch}
      />
    </NavigationProvider>
  )
}
