# DialogHeaderSearch

This component is mainly used in the [SelectDialog](/docs/components-dialog-selectdialog--default-story) to provide there an easy convenient solution of adding a search.

It leverages on the component [HeaderSearch](/docs/components-dialog-headersearch-headersearch--default-story) which provides the search logic.

export const Template = ({ title, iconLabel, ...args }) => {
  const [text, setText] = useState('')
  const clearSearch = () => setText('')
  return (
    <NavigationProvider>
      {
        <DialogHeaderSearch
          {...args}
          title={title}
          iconLabel={iconLabel}
          search={text}
          actions={{ search: setText, clear: clearSearch }}
          onDismiss={() => void 0}
        />
      }
    </NavigationProvider>
  )
}
