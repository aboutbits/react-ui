# SectionSearch

This component is used to add a standardized search input to the `SectionFilter` inside of a `SectionHeaderArea`.

export const Template = (args) => (
  <Section>
    <SectionHeaderArea>
      <SectionFilter initialValues={{ search: '' }} onSubmit={action('filter')}>
        <SectionSearch {...args} id="search" />
      </SectionFilter>
    </SectionHeaderArea>
  </Section>
)
