# SectionSearch

This component is used to add a standardized search input to the `FilterForm` inside of a `SectionHeaderArea`.

export const Template = (args) => (
  <Section>
    <SectionHeaderArea>
      <FilterForm defaultValues={{ search: '' }} onSubmit={action('filter')}>
        <SectionSearch {...args} id="search" />
      </FilterForm>
    </SectionHeaderArea>
  </Section>
)
