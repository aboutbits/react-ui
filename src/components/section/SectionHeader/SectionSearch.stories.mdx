# SectionSearch

This component is used to add a standardized search input to the `SectionFilter` inside of a `SectionHeader`.

export const Template = ({ ...args }) => (
  <Section>
    <SectionHeader>
      <SectionFilter initialValues={{ search: '' }} onFilter={action('filter')}>
        <SectionSearch {...args} />
      </SectionFilter>
    </SectionHeader>
  </Section>
)
