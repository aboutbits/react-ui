<Meta
  title="Components/React Hook Form/DateFormField"
  component={DateFormField}
  decorators={[
    (Story) => {
      const resolver = yupResolver(
        Yup.object({
          birthDate: Yup.date().required('Required'),
        })
      )
      const form = useForm({
        resolver,
        defaultValues: { birthDate: new Date() },
        mode: 'onTouched',
      })
      return (
        <Form form={form} onSubmit={action('onSubmit')}>
          <Story />
        </Form>
      )
    },
  ]}
/>

# DateFormField

An [InputField](../?path=/docs/components-form-inputfield--default-story) within the context of a `react-hook-form` form and with the default type `date`.

The form value that is returned for validation is of type `Date | null`. `null` is returned if the input is an empty string.
