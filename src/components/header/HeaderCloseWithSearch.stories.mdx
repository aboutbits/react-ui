# HeaderCloseWithSearch

This component is similar to [HeaderMainWithSearch](http://localhost:4000/?path=/docs/components-header-headermainwithsearch--default-story), but instead of the back button a close button is placed on the left side.

It leverages on the component [HeaderSearch](/docs/components-header-headersearch--default-story).

export const Template = ({ ...args }) => {
  const [text, setText] = useState('')
  const clearSearch = () => setText('')
  return (
    <NavigationProvider>
      <HeaderCloseWithSearch
        {...args}
        search={text}
        labelIcon="Open search"
        actions={{ search: setText, clear: clearSearch }}
        title="Header Title"
      />
    </NavigationProvider>
  )
}
