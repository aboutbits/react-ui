# MobileNavigationDialog

This component is a dialog for a mobile menu. It will cover the full app.
The close button of the menu is customizable. It makes use of the [MobileNavigationDialogCloseButton](/docs/components-navigation-MobileNavigationDialogCloseButton--default-story) component.

_**Important**_: This storybook is only available for a mobile view. Go to Canvas -> change the size of the preview -> Small mobile.

export const Template = ({ ...args }) => {
  return (
    <NavigationProvider>
      <div>
        <Header title="Home" />
        <MobileNavigationDialog {...args} dialogLabel="Mobile Menu">
          <Navigation navLabel={'Main Navigation'} className={'mt-14'}>
            <NavigationItemLink content={'Inbox'} icon={IconInbox} href={'/'} />
            <NavigationItemLink
              content={'Contacts'}
              icon={IconContacts}
              href={'/'}
            />
          </Navigation>
        </MobileNavigationDialog>
      </div>
    </NavigationProvider>
  )
}
